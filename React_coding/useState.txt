const ShowHidePassword = () => {

      </button>

6. Password Strength Meter

Show:

Password Strength

Weak 🔴

Medium 🟡

Strong 🟢

Rules:

< 6 characters → Weak
6–8 → Medium
8+ with numbers & symbols → Strong

import { useState } from "react";

const ShowHidePassword = () => {
  const [password, setPassword] = useState("");

  let strength = "";
  const hasUpper = /[A-Z]/.test(password);
  const hasNumber = /\d/.test(password);
  const hasSpChar = /[!@#$%^&*]/.test(password);

  if (password.length == 0) {
    strength = "";
  } else if (password.length < 6) {
    strength = "Weak";
  } else if (password.length < 8) {
    strength = "Medium";
  } else if (hasUpper && hasNumber && hasSpChar) {
    strength = "Strong";
  } else {
    strength = "Medium";
  }

  return (
    <div>
      <input
        type="password"
        placeholder="Enter password"
        value={password}
        onChange={(e) => setPassword(e.target.value)}
      />
      {strength && <p>Strength : {strength}</p>}
    </div>
  );
};

export default ShowHidePassword;

const ShowHidePassword = () => {

          <p>{hasNum ? "✅" : "❌"} Number</p>

          <p>{hasLength ? "✅" : "❌"} 8 Characters</p>
        </div>
      )}

  );
};

21. Search Number

Input

20

Found


Use map() when you want to change every element but keep the array length the same.
Use filter() when you want to remove elements based on a condition, so the array may become shorter.

Which should you use?
For debugging: JSON.stringify(num)
To display an array like [10, 20, 30]: [{num.join(", ")}]
For real React UIs: Use .map() to render elements (lists, cards, table rows, etc.).
Interview Tip

If asked "Why doesn't React show the brackets?", you can answer:

React can render arrays of values directly. When an array is rendered inside JSX, it is converted to a comma-separated string (for primitive values like numbers and strings), so the square brackets are not displayed. If you want brackets, you need to format the output yourself, for example with JSON.stringify() or join().
